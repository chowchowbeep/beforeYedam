조건문(conditional statement)은 javascript에서 중요한 내용. 비교를 이용할 것임. example) if //ex 1) if(true){ alert('result : true'); // result : true 가 경고창으로 표시 } //ex 2) if(false){ alert('result : true'); // 아무것도 표시 안됨 } //ex 3) if(true){ alert(1); alert(2); alert(3); alert(4); } alert(5); //1,2,3,4,5 경고창이 차례로 출력 //ex 4) if(false){ alert(1); alert(2); alert(3); alert(4); } alert(5); //5만 경고창으로 출력.
